#include <Servo.h>
Servo servoRight;
Servo servoLeft;

void forwardM(int time){
  servoRight.writeMicroseconds(1300);
  servoLeft.writeMicroseconds(1700);
  delay(time);
}

void stopM(){
  servoLeft.writeMicroseconds(1500);
  servoRight.writeMicroseconds(1500);
}

void backM(){
 servoRight.writeMicroseconds(1700);
 servoLeft.writeMicroseconds(1300); 
}

void rightM(){
  servoRight.writeMicroseconds(1500);
  servoLeft.writeMicroseconds(1700);
}

void leftM(){
  servoRight.writeMicroseconds(1300);
  servoLeft.writeMicroseconds(1500);
}

void setup(){
  //Serial.begin(9600);
  servoLeft.attach(13);
  servoRight.attach(12);
  servoLeft.writeMicroseconds(1500);
  servoRight.writeMicroseconds(1500);
  //delay(5000);
  //Serial.print("this doesnâ€™t make a new line");
  //Serial.println("this does");
  //Serial.println(Left);
  //Serial.println(Right);
}
int counter=0;

void loop(){
byte Left= digitalRead(5); //outputs HIGH or LOW
byte Right= digitalRead(7); 

if (Left==1 && Right==1){
  forwardM(2000);
}

if (Left==0){
  backM();
  delay(1000);
  leftM();
  delay(1500);
  counter++;
}

if (Right==0){
  backM();
  delay(1000);
  rightM();
  delay(1500);
  counter++;
}

if (Left==0 and Right==0){
  backM();
  delay(1000);
  rightM();
  delay(1500);
  counter++;
}

if(counter>4){
  backM();
  delay(1200);
  leftM();
  delay(1700);

}

}
